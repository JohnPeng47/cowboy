    def test_repo_push_branch_deleted_non_existent(self):
        branch_name = "non-existent-branch"
        response = self._post_event_data(
            event=BitbucketWebhookEvents.REPO_PUSH,
            data={
                "repository": {"uuid": "{673a6070-3421-46c9-9d48-90745f7bfe8e}"},
                "push": {
                    "changes": [
                        {
                            "new": None,
                            "old": {
                                "type": "branch",
                                "name": branch_name,
                                "target": {},
                            },
                            "links": {},
                            "created": False,
                            "forced": False,
                            "closed": True,
                            "commits": [],
                            "truncated": False,
                        }
                    ]
                },
            },
        )
        assert response.status_code == status.HTTP_200_OK
        assert response.data is None
        assert not Branch.objects.filter(
            repository=self.repo, name=branch_name
        ).exists()
