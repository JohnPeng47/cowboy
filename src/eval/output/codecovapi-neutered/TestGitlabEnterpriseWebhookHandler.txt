    def test_job_event_commit_complete_but_repo_deactivated(self):

        commit = CommitFactory(

        )
        self.repo.active = False
        self.repo.save()
        response = self._post_event_data(

            },
        )

    def test_system_hook_project_update_no_owner_token(self):
        owner = OwnerFactory(service="gitlab_enterprise", oauth_token=None)
        repo = RepositoryFactory(author=owner, service_id=uuid.uuid4(), active=True)
        data = {
            "event_name": "project_update",
            "project_id": repo.service_id,
            "name": "UpdatedProjectName",
            "path_with_namespace": f"{owner.username}/updatedprojectname",
        }
        response = self._post_event_data(event=GitLabWebhookEvents.SYSTEM, data=data)
        assert response.status_code == status.HTTP_404_NOT_FOUND

    def test_merge_request_event_unsupported_action(self):
        pullid = 3
        unsupported_action = "reopened"
        response = self._post_event_data(
            event=GitLabWebhookEvents.MERGE_REQUEST,
            data={
                "object_kind": "merge_request",
                "object_attributes": {
                    "action": unsupported_action,
                    "target_project_id": self.repo.service_id,
                    "iid": pullid,
                },
            },
        )
        assert response.status_code == status.HTTP_200_OK
        assert "Unhandled Gitlab webhook merge_request action" in str(response.data)
